# 数据类型定义

以下为前端对接所需的完整 DTO 与领域模型结构，按接口出现顺序组织。

---

## 1. 目标相关

### CreateGoalRequest（请求）

```ts
interface CreateGoalRequest {
  rawGoalText: string;
  timeBudget?: TimeBudget;
  selfReportedLevel?: SelfReportedLevel;
  preferenceTags?: PreferenceTag[];
  goalTypeHint?: GoalType;
  subjectHint?: string;
  topicHints?: string[];
  sourceContext?: string;
  priorityModule?: string;
}
```

### CreateGoalData（响应）

```ts
interface CreateGoalData {
  goalId: string;
  structuredGoal: StructuredLearningGoal;
  goalContextSnapshot: GoalContextSnapshot;
}
```

### StructuredLearningGoal

```ts
interface StructuredLearningGoal {
  rawGoalText: string;
  normalizedGoalText?: string;
  goalType: GoalType;
  subject?: string;
  topicScopeType?: string;
  topics?: string[];
  intentDescription?: string;
  timeBudget?: TimeBudget;
  urgencyLevel?: UrgencyLevel;
  expectedDepth?: string;
  selfReportedLevel?: SelfReportedLevel;
  preferenceTags?: PreferenceTag[];
  constraints?: string[];
  sourceContext?: string;
  priorityModule?: string;
}
```

### GoalContextSnapshot

```ts
interface GoalContextSnapshot {
  structuredGoal?: StructuredLearningGoal;
  requiresDiagnosis: boolean;
  planningMode: PlanningMode;
  entryGranularity: EntryGranularity;
  strategyHints?: string[];
  riskTags?: string[];
  explanationFocus?: string[];
  createdFrom?: string;
  version?: number;
}
```

## 2. 诊断相关

### CreateDiagnosisSessionRequest（请求）

```ts
interface CreateDiagnosisSessionRequest {
  goalId: string;
}
```

### DiagnosisSessionData（响应）

```ts
interface DiagnosisSessionData {
  diagnosisId: string;
  sessionId: string;
  status: string;  // DiagnosisSessionStatus
  generationMode: string;
  questions: DiagnosisQuestion[];
}
```

### DiagnosisQuestion

```ts
interface DiagnosisQuestion {
  questionId: string;
  dimension: string;
  type: string;    // SINGLE_CHOICE / MULTI_CHOICE 等
  required: boolean;
  title: string;
  description?: string;
  whyAsking?: string;
  impactsPlanning?: string[];
  options: DiagnosisOption[];
}
```

### DiagnosisOption

```ts
interface DiagnosisOption {
  code: string;
  label: string;
  order?: number;
}
```

### SubmitDiagnosisRequest（请求）

```ts
interface SubmitDiagnosisRequest {
  diagnosisId: string;
  answers: DiagnosisAnswer[];
}

interface DiagnosisAnswer {
  questionId: string;
  selectedOptions: string[];  // 选项 code 列表
  textAnswer?: string;
}
```

### SubmitDiagnosisData（响应）

```ts
interface SubmitDiagnosisData {
  diagnosisId: string;
  learnerProfileSnapshot: LearnerProfileSnapshot;
  diagnosisEvidenceSummary: DiagnosisEvidenceSummary;
}
```

### LearnerProfileSnapshot

```ts
interface LearnerProfileSnapshot {
  diagnosisId: string;
  foundationLevel: FoundationLevel;
  executionStability?: ExecutionStability;
  timeBudgetLevel?: TimeBudget;
  learningPreference?: LearningPreference;
  blockingPoint?: string;
  urgencyLevel?: UrgencyLevel;
  blockerTags?: string[];
  riskTags?: string[];
}
```

### DiagnosisEvidenceSummary

```ts
interface DiagnosisEvidenceSummary {
  summary?: string;
  keyEvidence?: string[];
  primaryGapType?: string;
  primaryRiskTags?: string[];
  explanationPoints?: string[];
}
```

## 3. 规划相关

### PreviewLearningPlanRequest（请求）

```ts
interface PreviewLearningPlanRequest {
  goalId: string;
  diagnosisId: string;
}
```

### PlanPreviewData（响应）

```ts
interface PlanPreviewData {
  planId: string;
  status: string;  // PlanStatus
  previewOnly: boolean;
  committed: boolean;
  goal?: string;
  recommendedEntry: RecommendedEntry;
  recommendedStrategy: RecommendedStrategy;
  stages: PlanStage[];
  tasks: TaskBlueprint[];
  successCriteria?: string[];
  keyEvidence?: string[];
  risks?: string[];
}
```

### RecommendedEntry

```ts
interface RecommendedEntry {
  conceptId?: string;
  title: string;
  estimatedMinutes?: number;
  reason?: string;
}
```

### RecommendedStrategy

```ts
interface RecommendedStrategy {
  code: RecommendedStrategyCode;
  label?: string;
  reason?: string;
}
```

### PlanStage

```ts
interface PlanStage {
  stageCode: string;
  title: string;
  objective?: string;
  estimatedMinutes?: number;
}
```

### TaskBlueprint

```ts
interface TaskBlueprint {
  taskId: string;
  title: string;
  taskType: TaskType;
  goal: string;
  taskMethod?: string;
  recommendedPromptTemplate?: string;
  estimatedMinutes?: number;
  promptScaffold?: string;  // deprecated，优先用 recommendedPromptTemplate
  completionCriteria?: string[];
  evidenceToCollect?: string[];
  selfEvaluationQuestions?: string[];
  fallbackAction?: string;
}
```

### CommitLearningPlanRequest（请求）

```ts
interface CommitLearningPlanRequest {
  planId: string;
}
```

### CommitPlanData（响应）

```ts
interface CommitPlanData {
  sessionId: string;
  planId: string;
  taskSequence: string[];
  currentTaskId: string;
  status: string;  // LearningSessionStatus
}
```

## 4. 任务执行相关

### CurrentTaskData（响应）

```ts
interface CurrentTaskData {
  sessionId: string;
  currentTask: CurrentTaskItem;
  progress: ProgressItem;
}

interface CurrentTaskItem {
  taskId: string;
  title: string;
  taskType: string;
  goal: string;
  whyThisTask?: string;
  taskMethod?: string;
  recommendedPromptTemplate?: string;
  estimatedMinutes?: number;
  promptScaffold?: string;
  completionCriteria?: string[];
  selfEvaluationQuestions?: string[];
  fallbackAction?: string;
}

interface ProgressItem {
  currentIndex: number;
  totalTasks: number;
}
```

### TaskInteractionRequest（请求）

```ts
interface TaskInteractionRequest {
  sessionId: string;
  interactionType?: string;
  interactionCountDelta?: number;
  userSummarySubmitted?: boolean;
  behaviorSignals?: string[];
  detectedIssueTags?: string[];
  contentSummary?: string;
}
```

### TaskInteractionData（响应）

```ts
interface TaskInteractionData {
  taskId: string;
  accepted: boolean;
}
```

### CompleteTaskRequest（请求）

```ts
interface CompleteTaskRequest {
  sessionId: string;
  completionStatus: TaskCompletionStatus;
  durationMinutes?: number;
  interactionCount?: number;
  userSummarySubmitted?: boolean;
  microPracticeResult?: string;
  detectedIssueTags?: string[];
  behaviorSignals?: string[];
  learnerReflection?: string;
}
```

### CompleteTaskData（响应）

```ts
interface CompleteTaskData {
  taskExecutionRecord: TaskExecutionRecord;
  nextTaskAvailable: boolean;
  nextTaskId?: string;
  sessionProgress?: SessionProgressItem;
}

interface TaskExecutionRecord {
  taskId: string;
  taskType: string;
  completionStatus: string;
  durationMinutes?: number;
  interactionCount?: number;
  userSummarySubmitted?: boolean;
  microPracticeResult?: string;
  detectedIssueTags?: string[];
  behaviorSignals?: string[];
  learnerReflection?: string;
}

interface SessionProgressItem {
  completedTasks: number;
  totalTasks: number;
}
```

## 5. 报告与 Next Action

### ReportData（响应）

```ts
interface ReportData {
  learningReport: LearningReport;
  nextActionDecision: NextActionDecision;
}
```

### LearningReport

```ts
interface LearningReport {
  sessionId: string;
  resultStatus: ResultStatus;
  goalReview?: string;
  completedProgress?: string[];
  unresolvedIssues?: string[];
  evidenceSummary?: string[];
  summaryText?: string;
  nextAction?: NextActionDecision;
}
```

### NextActionDecision

```ts
interface NextActionDecision {
  actionType: NextActionType;
  reason?: string;
  nextEntryPoint?: string;
  adjustmentSignals?: string[];
  requiresReplan: boolean;
}
```

### ConfirmNextActionRequest（请求）

```ts
interface ConfirmNextActionRequest {
  actionType: NextActionType;
}
```

### NextActionConfirmData（响应）

```ts
interface NextActionConfirmData {
  sessionId: string;
  acceptedAction: NextActionType;
  requiresReplan: boolean;
  nextHint?: string;
}
```
